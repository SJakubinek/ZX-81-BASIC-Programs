10 PRINT "DENKE DIR EINE ZAHL ZWISCHEN 1  UND 10."
20 LET A=1
30 LET B=10
40 LET C=INT ((RND*(B-A+1))+A)
50 PRINT "MEINE ZAHL: ";C

70 INPUT Z$
80 IF Z$="H" THEN GOTO 500
90 IF Z$="N" THEN GOTO 1000
100 IF Z$="G" THEN PRINT "JUHU"
110 STOP             
500 LET B=C
510 GOTO 40
1000 LET A=C
1010 GOTO 40
